  <action>
    Create src/app/api/v1/weather/current/route.ts:

    export async function GET(request: Request) {
      // 1. Rate limit
      const { allowed } = checkRateLimit(request)
      if (!allowed) return Response.json({ error: 'Rate limit exceeded', code: 1002 }, { status: 429 })

      // 2. Validate input
      const { searchParams } = new URL(request.url)
      const city = validateCityParam(searchParams.get('city'))
      if (!city) return Response.json({ error: 'Invalid city parameter', code: 1004 }, { status: 400 })

      // 3. Cache check
      const cached = cache.get(city)
      if (cached) return Response.json(cached, { status: 200 })

      // 4. Fetch upstream
      try {
        const data = await fetchCurrentWeather(city)
        cache.set(city, data)
        return Response.json(data, { status: 200 })
      } catch (err: any) {
        const code: number = err.code ?? 1003
        const status = code === 1001 ? 404 : code === 1002 ? 429 : 503
        return Response.json({ error: err.message ?? 'Upstream error', code }, { status })
      }
    }

    Imports: checkRateLimit, validateCityParam, cache (get/set), fetchCurrentWeather.
    Error code mapping: 1001→404, 1002→429, 1003→503.
  </action>
  <verify>
    `curl -s "http://localhost:3000/api/v1/weather/current?city=London"` returns WeatherData with temp_c and temp_f.
    `curl -s "http://localhost:3000/api/v1/weather/current?city=a"` returns 400 code 1004.
    `curl -s "http://localhost:3000/api/v1/weather/current?city=London" | jq .icon_url` starts with "https://".
    Second identical request within 60s is served from cache (check server logs — no second weatherapi call).

<task id="T9" type="auto">
  <title>Multiple Cities Endpoint</title>
  <files>src/app/api/v1/weather/multiple/route.ts</files>
  <action>
    Create src/app/api/v1/weather/multiple/route.ts:

    export async function GET(request: Request) {
      // 1. Rate limit
      const { allowed } = checkRateLimit(request)
      if (!allowed) return Response.json({ error: 'Rate limit exceeded', code: 1002 }, { status: 429 })

      // 2. Validate cities param
      const { searchParams } = new URL(request.url)
      const rawCities = searchParams.get('cities')
      const cities = validateCitiesParam(rawCities)
      if (!cities) {
        // Distinguish 1004 vs 1005 — validateCitiesParam returns null for both; check count manually
        const rawCount = rawCities?.split(',').map(s => s.trim()).filter(Boolean).length ?? 0
        const code = rawCount > 10 ? 1005 : 1004
        const status = 400
        return Response.json({ error: code === 1005 ? 'Max cities exceeded' : 'Invalid cities parameter', code }, { status })
      }

      // 3. Fetch all cities in parallel; each hit cache or upstream
      const results = await Promise.allSettled(
        cities.map(async (city): Promise<MultiWeatherResult> => {
          const cached = cache.get(city)
          if (cached) return cached
          try {
            const data = await fetchCurrentWeather(city)
            cache.set(city, data)
            return data
          } catch (err: any) {
            return { type: 'error', city, error: err.message ?? 'Failed', code: err.code ?? 1003 }
          }
        })
      )

      // 4. Map settled results — always 200 (partial success model)
      const response: MultiWeatherResult[] = results.map((r, i) =>
        r.status === 'fulfilled'
          ? r.value
          : { type: 'error', city: cities[i], error: 'Failed to fetch', code: 1003 }
      )

      return Response.json(response, { status: 200 })
    }

    Note: do NOT call the HTTP /weather/current route internally — call fetchCurrentWeather() and cache directly to avoid double rate-limit accounting.
    Imports: checkRateLimit, validateCitiesParam, cache, fetchCurrentWeather, MultiWeatherResult.
  </action>
  <verify>
    `curl -s "http://localhost:3000/api/v1/weather/multiple?cities=London,Paris"` returns array length 2.
    `curl -s "http://localhost:3000/api/v1/weather/multiple?cities=London,XXXXXXXXFAKECITY"` returns 200 with one WeatherData and one WeatherError (type:'error').
    `curl -s "http://localhost:3000/api/v1/weather/multiple?cities=$(python3 -c "print(','.join(['city']*11))")"` returns 400.
    `curl -s "http://localhost:3000/api/v1/weather/multiple?cities=London,,Paris"` returns 400 code 1004.
  </verify>
  <done>Multiple cities route returns 200 MultiWeatherResult[] with partial success. 400 for invalid params. Length matches deduped city count.</done>
</task>
