    ```typescript
    import { checkRateLimit } from '@/lib/rate-limit'
    import { validateSearchQuery } from '@/lib/validation'
    import { getDb } from '@/lib/db'
    import type { CityEntry } from '@/types/weather'

    export async function GET(request: Request) {
      const { allowed } = checkRateLimit(request)
      if (!allowed) return Response.json({ error: 'Rate limit exceeded', code: 1002 }, { status: 429 })

      const { searchParams } = new URL(request.url)
      const q = validateSearchQuery(searchParams.get('q'))
      if (!q) return Response.json({ error: 'Invalid query parameter', code: 1004 }, { status: 400 })

      const db = getDb()
      const rows = db
        .prepare(
          `SELECT id, name, country, region, lat, lon, timezone
           FROM cities
           WHERE name LIKE ?
           ORDER BY population DESC
           LIMIT 8`
        )
        .all(q + '%') as CityEntry[]

      return Response.json(rows, { status: 200 })
    }
    ```

    CRITICAL — SQL injection prevention:
    - The SQL string is a static string literal with a single `?` positional parameter
    - The user input (q) is passed as the `.all()` argument, NEVER concatenated into the SQL string itself
    - The `%` wildcard is appended to q in JavaScript (q + '%'), not inside the SQL — this is correct and safe

    CRITICAL — Do NOT import from '@/lib/weatherapi' or '@/lib/cache' — this route no longer uses weatherapi.com
    CRITICAL — The prepared statement selects exactly: id, name, country, region, lat, lon, timezone (matches CityEntry interface exactly — do NOT include city_ascii or population in the SELECT)

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Injection | /api/v1/cities/search — `?q=` parameter | mitigate | validateSearchQuery() rejects any input not matching /^[a-zA-Z0-9 ,'.\\-]+$/; parameterized LIKE ? prevents SQL injection even if validation were bypassed — defense in depth |
| T-03-02 | Information Disclosure | cities/search returning internal DB errors | mitigate | Route catches no DB errors explicitly — better-sqlite3 throws synchronously; Next.js catches unhandled errors and returns 500 without DB internals in production mode |
| T-03-03 | Denial of Service | Unbounded result set from LIKE query | mitigate | LIMIT 8 hard-coded in SQL — cannot be overridden by user input |
| T-03-04 | Denial of Service | Rate limit bypass via search endpoint | mitigate | checkRateLimit() applied before any DB access — identical to all other routes |
| T-03-05 | Tampering | Modification of CityEntry data via search route | accept | getDb() opens cities.db with { readonly: true } — write operations throw immediately; search route is read-only by design |
